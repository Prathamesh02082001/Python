import re

Q.3)Create a program to assess the strength of a user-provided password. Use a combination of regular expressions and data structures to evaluate the password's length, complexity, and presence of special characters. Assign a strength rating (e.g., weak, medium, strong) based on the evaluation.
code:
def passw(password):
    pass_pattern = re.compile(r"[a-zA-Z0-9_.]+[@#$%^&*]+[a-zA-Z0-9_.]+")
    length = re.compile(r".{9,}")

    if pass_pattern.match(password):
        print("Valid")
    else:
        print("Invalid")

    if length.match(password) and len(password) >= 9:
        print("Hard")
    elif len(password) == 9:
        print("Medium")
    else:
        print("Simple")

passw("Rocco@1299")
